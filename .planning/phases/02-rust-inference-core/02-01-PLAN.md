    ```rust
    /// Rust-internal error type for the GTE inference pipeline.
    /// Phase 3 will convert these into magnus::Error / GTE::Error (Ruby).
    #[derive(Debug)]
    pub enum GteError {
        /// Tokenizer initialization or encoding failure
        Tokenizer(String),
        /// ORT session creation or inference failure
        Inference(String),
        /// ORT library error (wrapped from ort::Error)
        Ort(String),
        /// Array dimensionality mismatch during embedding extraction
        Shape(String),
    }

    impl std::fmt::Display for GteError {
        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
            match self {
                GteError::Tokenizer(msg) => write!(f, "GTE tokenizer error: {}", msg),
                GteError::Inference(msg) => write!(f, "GTE inference error: {}", msg),
                GteError::Ort(msg) => write!(f, "GTE ORT error: {}", msg),
                GteError::Shape(msg) => write!(f, "GTE shape error: {}", msg),
            }
        }
    }

    impl std::error::Error for GteError {}

    impl From<ort::Error> for GteError {
        fn from(e: ort::Error) -> Self {
            GteError::Ort(e.to_string())
        }
    }

    impl From<ndarray::ShapeError> for GteError {
        fn from(e: ndarray::ShapeError) -> Self {
            GteError::Shape(e.to_string())
        }
    }

    /// Crate-level Result alias — use `crate::Result<T>` throughout the inference modules
    pub type Result<T> = std::result::Result<T, GteError>;
    ```

    Do NOT add impl From<magnus::Error> — that conversion goes in Phase 3 when the Ruby FFI layer is added.

    /// Extraction mode for pulling embeddings from ORT session output.
    /// Per D-04: simple enum with match arms, no trait objects.
    #[derive(Debug, Clone, Copy)]
    pub enum ExtractorMode {
        /// Output tensor shape is [batch, seq, dim] — take token at `index` (0 = CLS token)
        Token(usize),
        /// Output tensor shape is [batch, dim] — use as-is (already pooled)
        Raw,
    }

    /// Per-model-family configuration that drives tokenization and embedding extraction.
    /// Per D-03: plain struct with factory methods, not a trait.
    #[derive(Debug, Clone)]
    pub struct ModelConfig {
        /// Maximum token length — inputs are truncated here (RUST-05)
        pub max_length: usize,
        /// Name of the output tensor in the ONNX graph to extract embeddings from
        pub output_tensor: &'static str,
        /// How to extract the embedding vector from the output tensor
        pub mode: ExtractorMode,
        /// Whether the model expects a token_type_ids input tensor
        pub with_type_ids: bool,
    }

    impl ModelConfig {
        /// E5 model family defaults:
        /// - max_length: 512 tokens
        /// - output_tensor: "last_hidden_state" (rank-3: [batch, seq, dim])
        /// - mode: Token(0) — CLS token extraction
        /// - with_type_ids: true — E5 BERT-based models need token_type_ids
        pub fn e5() -> Self {
            Self {
                max_length: 512,
                output_tensor: "last_hidden_state",
                mode: ExtractorMode::Token(0),
                with_type_ids: true,
            }
        }

        /// CLIP model family defaults:

        /// - with_type_ids: false — CLIP does not use token_type_ids

            Self {
                max_length: 77,
                output_tensor: "text_embeds",
                mode: ExtractorMode::Raw,
