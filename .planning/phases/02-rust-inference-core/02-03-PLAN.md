    ```rust
    use std::path::Path;
    use ndarray::Array2;
    use ort::session::Session;
    use crate::model_config::ModelConfig;
    use crate::tokenizer::Tokenizer;
    use crate::session::{build_session, run_session};

    /// The main inference orchestrator for Phase 2.
    /// Holds all initialized state: tokenizer, ORT session, model config.
    /// Phase 3 will wrap this in a #[wrap] magnus struct for Ruby FFI.
    pub struct Embedder {
        tokenizer: Tokenizer,
        session: Session,
        config: ModelConfig,
    }

    impl Embedder {
        /// Initialize the embedder from file paths and a model config.
        ///
        /// # Arguments
        /// - `tokenizer_path`: path to tokenizer.json
        /// - `model_path`: path to model.onnx
        /// - `config`: ModelConfig with max_length, output_tensor, mode, with_type_ids
        ///
        /// # Example
        /// ```no_run
        /// let config = ModelConfig::e5();
        /// let embedder = Embedder::new("path/to/tokenizer.json", "path/to/model.onnx", config)?;
        /// ```
        pub fn new<P1, P2>(
            tokenizer_path: P1,
            model_path: P2,
            config: ModelConfig,
        ) -> crate::Result<Self>
        where
            P1: AsRef<Path>,
            P2: AsRef<Path>,
        {
            let tokenizer = Tokenizer::new(
                tokenizer_path,
                config.max_length,
                config.with_type_ids,
            )?;
            let session = build_session(model_path)?;
            Ok(Self { tokenizer, session, config })
        }

        /// Tokenize a batch of strings and run the ONNX inference session.
        /// Returns a 2D array of shape [batch_size, embedding_dim].
        ///
        /// RUST-03: Embedding extraction mode is determined by self.config.mode.
        pub fn embed(&self, texts: Vec<String>) -> crate::Result<Array2<f32>> {
            let tokenized = self.tokenizer.tokenize(texts)?;
            let embeddings = run_session(&self.session, &tokenized, &self.config)?;
            Ok(embeddings)
        }
    }
    ```

    Key design notes:
    - `Embedder::new()` extracts `max_length` and `with_type_ids` from `config` before moving `config` into the struct — this is why Tokenizer::new takes these as separate params (config is moved into self.config)
    - `embed()` returns owned `Array2<f32>` — session outputs are extracted and dropped inside `run_session`
    - No L2 normalization in Phase 2 — deferred to Phase 3 per CONTEXT.md discretion note. Add a code comment: `// L2 normalization is applied in Phase 3 (Ruby API layer)`

    Then update ext/gte/src/lib.rs — add this line after the existing module declarations:

    use gte::tokenizer::Tokenizer;

    #[test]
    #[ignore = "requires ext/gte/tests/fixtures/e5/tokenizer.json"]
    fn test_e5_tokenizer_output_shape() {
        const TOKENIZER: &str = concat!(
            env!("CARGO_MANIFEST_DIR"),
            "/tests/fixtures/e5/tokenizer.json"
        );

        // E5 max_length = 512, with_type_ids = true
        let tokenizer = Tokenizer::new(TOKENIZER, 512, true)
            .expect("tokenizer should load from fixture");

        let texts = vec![
            "Hello, world!".to_string(),
            "A second, longer sentence to test padding behavior.".to_string(),
        ];

        let tokenized = tokenizer.tokenize(texts).expect("tokenize should succeed");

        // Both rows must have the same column count (BatchLongest padding)
        assert_eq!(
            tokenized.input_ids.shape()[0], 2,
            "batch size should be 2"
        );
        assert_eq!(
            tokenized.attn_masks.shape()[0], 2,
            "attention mask batch size should be 2"
        );
        assert_eq!(
            tokenized.input_ids.shape()[1], tokenized.attn_masks.shape()[1],
            "input_ids and attn_masks must have same sequence length (BatchLongest)"
        );
        assert!(
            tokenized.type_ids.is_some(),
            "E5 config with_type_ids=true should produce type_ids"
        );
    }

    #[test]
    #[ignore = "requires ext/gte/tests/fixtures/e5/tokenizer.json"]
    fn test_e5_truncation_at_max_length() {
        const TOKENIZER: &str = concat!(
            env!("CARGO_MANIFEST_DIR"),
            "/tests/fixtures/e5/tokenizer.json"
        );

        // Use max_length=16 to force truncation of a long input

            "sequence length must be truncated to max_length=16 (RUST-05)"
